<script>
const products = {
  "iphone 11": 3000000,
  "iphone 12": 4500000,
  "iphone 13": 6500000,
  "iphone 14": 8500000
};

// SEARCH PRODUK
document.getElementById("search").addEventListener("input", function() {
  let value = this.value.toLowerCase();
  let result = document.getElementById("result");

  if (products[value]) {
    let harga = products[value].toLocaleString("id-ID");

    result.innerHTML = `
      <div class="card">
        <h2>${value.toUpperCase()}</h2>
        <p>💰 Rp ${harga}</p>
        <a class="btn" href="https://wa.me/628XXXXXXXXXX?text=Saya%20mau%20order%20${value}%20harga%20Rp%20${harga}">
          🛒 Beli via WhatsApp
        </a>
      </div>
    `;
  } else {
    result.innerHTML = "<p>Produk tidak ditemukan</p>";
  }
});

// AI SEDERHANA (AUTO JAWAB)
document.getElementById("question").addEventListener("input", function() {
  let q = this.value.toLowerCase();
  let answer = document.getElementById("answer");

  if (q.includes("harga")) {
    answer.innerHTML = "💰 Harga tergantung produk, ketik nama produk di atas ya.";
  }
  else if (q.includes("ori")) {
    answer.innerHTML = "✅ Produk kami original & bergaransi.";
  }
  else if (q.includes("cod")) {
    answer.innerHTML = "🚚 Bisa COD di area tertentu.";
  }
  else if (q.includes("lama pengiriman")) {
    answer.innerHTML = "📦 Pengiriman 1-3 hari kerja.";
  }
  else {
    answer.innerHTML = "🤖 Admin akan segera membalas pertanyaan kamu.";
  }
});
</script>
